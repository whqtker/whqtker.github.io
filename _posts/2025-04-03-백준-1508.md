#include <algorithm>
#include <vector>

using namespace std;

int main() {
	cin.tie(0);
	ios::sync_with_stdio(0);

		v.push_back(x);

	int l = 0;
	int r = v[n - 1];

	while (l <= r) {
		int cnt = 1;

			}
		}

			ans = max(ans, m);
			l = m + 1;
		}
		else {
			r = m - 1;
		}
	}

	cout << ans;
}
